plus_x = [0, 1, 0, -1] # 왼쪽 아래 오른쪽 위

def killFly(N, my_list, row, col):
    '''
    리스트의 row, col 에서
    + 방향, X방향 중 더 많은 파리를 죽이는 방향의 개수 리턴 
    '''
    # + 로 죽이는 파리수
    count1 = 0 
    count1 += my_list[row][col]
    for l in range(4):
        for k in range(1, M):
            x = row + plus_x[l] * k
            y = col + plus_y[l] * k
            if (x >= 0 and x < N) and (0 <= y < N): # 예외 처리
                count1 += my_list[x][y]
# x 로 죽이는 파리수
count2 = my_list[row][col]
for l in range(4):
    for k in range(1, M):
        x = row + mul_x[l] * k
        y = col + mul_y[l] * k
        if (x >= 0 and x < N) and (0 <= y < N):
            count2 += my_list[x][y]

# +, X 로 죽일 수 있는 파리 중 가장 큰 값 리턴
if count1 > count2:
