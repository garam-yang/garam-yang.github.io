class Solution(object):
    def numIslands(self, grid):

        if not grid:
            return 0
        count = 0
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j]=='1':
                # By the condition, grid's elements are 'string'type.
                    self.dfs(grid, i, j)
                    count+=1
        return count
